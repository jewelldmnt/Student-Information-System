[VECTOR ERASE]

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {

    int N, i, rem, rem1, rem2;
    cin >> N;

    vector<int>v(N);

    for (i = 0; i < N; i++){
        cin >> v[i];
    }

    cin >> rem;

    cin >> rem1 >> rem2;

    int x = rem - 1;
    v.erase(v.begin()+x);

    int a = rem1 - 1, b = rem2 - 1;
    v.erase(v.begin()+a,v.begin()+b);

    int length = v.size();
    cout << length << endl;

    for (i = 0; i < length; i++){
        cout << v[i] << " ";
    }

    return 0;
}
